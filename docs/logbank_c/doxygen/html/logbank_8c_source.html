273update_header(char* header, size_t size, int level, log_locate_t locate) {
274 if (header == NULL) {
275 /* header doesn't have a dedicated memory */
276 return 0;
277 }
278
279 int len = 0;
280
281 /* dim colors of stamps */
282 len += LOG_ADD_COLOR(header + len, size - len, LOG_COLOR_GRAY);
283
284 /* log datestamps */

286 char date_stamp[LOG_DATE_LEN];

295 log_stamp_collector(time_stamp, LOG_TIME_FORMAT, sizeof(time_stamp));
296
297 len += snprintf(header + len, size - len, "%s ", time_stamp);
298 }
299
300 /* use pre-defined color for level names */
301 len += LOG_ADD_COLOR(header + len, size - len, level);
302
303 /* Reading name of requested priority-level */
304 const char* name = log_get_levelname(level);
305
306 /* log priority-level name */
307 if (level == LOG_INFO || level == LOG_WARN) {
308 len += snprintf(header + len, size - len, "[%s] ", name);
309 } else {
310 len += snprintf(header + len, size - len, "[%s] ", name);
311 }
312
313 /* log location parameters */
314 if (locate.func != NULL && locate.file != NULL) {
315
316 /* dim color of location parameters and revert to color of log priority-level */
317 len += LOG_ADD_COLOR(header + len, size - len, LOG_COLOR_GRAY);
318 len += snprintf(header + len, size - len, "@ %s() \"%s +%d\" ", locate.func,
319 locate.file, locate.line);
320
321 len += LOG_ADD_COLOR(header + len, size - len, level);
322 }
323
324 /* log separator between header (message-prefix) and message */
325 len += snprintf(header + len, size - len, ": ");
326
327 /* terminate the use of colors */
328 len += LOG_ADD_COLOR(header + len, size - len, LOG_COLOR_END);

352log_en_queue(const char* message) {
353 int read_index = log_queue.read_index;
354 int write_index = log_queue.write_index;
355 int queue_size = LOG_RETRIEVE_QUEUE_LEN + 1;
356
357 if ((write_index + 1) % queue_size == read_index) {
358 /* Queue Full */

360 }

363 log_queue.write_index = (write_index + 1) % queue_size;

365 size_t size = LOG_MAX_DATA_LEN;

592parse_level(const char* copy, log_parser_info_t* info) {
593 int invalid = LOG_FATAL + 1;

769static int
770parse_data_string(char* message, size_t size, const char* copy) {
771 if (copy == NULL) {
772 /* No data available to parse */
773 return LOG_PARSER_FAILED;
774 }
775
776 /*
777 * Calculating Data Index
778 * HH:MM:SS [level] @ location : data
779 * 0 1 2
780 * */
781 int data_index = 2;
782
783 /* Separating header from copied data */
784 const char* dptr = strchr(copy, ':');
785 for (int count = 0; count < data_index && dptr != NULL; count++) {
786 dptr = strchr(dptr + 1, ':');
787 }
788
789 if (dptr == NULL) {
790 /* Time stamp is absent; Set data pointer to first ':' */
791 dptr = strchr(copy, ':');
792 }
793
794 strncpy(message, dptr + 2, size);
795 message[size - 1] = '\0';
796 return LOG_PARSER_DONE;
797}

875int
876log_this(int level, log_locate_t locate, const char* fmt, ...) {
877 int len = 0;
878
879 /* Filtering logic for lower priority log-requests and invalid priority requests */
880 if (level < log_level || level > LOG_FATAL) {
881 return len;
882 }
883
884 /* prepare log-message */
885 char message[LOG_MAX_DATA_LEN];
886 size_t size = sizeof(message);
887
888 /* Add header to log-message */
889 len += update_header(message, size - len, level, locate);
890
891 /* Add optional arguments to log-message */
892 va_list args;
893 va_start(args, fmt);
894 len += vsnprintf(message + len, size - len, fmt, args);
895 va_end(args);
896
897 /* Handle string termination properly */
898 len += snprintf(message + len, size - len, "\n");
899 message[size - 1] = '\0';
900
901 if (len > LOG_MAX_DATA_LEN) {
902 /* Data Overflow */
903 return (0 - len);
904 }
905
906 /* write operation */
907 log_stream_write(message);
908 log_en_queue(message);
909 return len;
910}

936int
937log_parser(char* message, size_t size, log_parser_info_t* info) {
938 if (message == NULL || size < LOG_MAX_DATA_LEN || info == NULL) {
939 /* Invalid parameters */
940 return -1;
941 }
942
943 /* Create a reference copy */
944 char copy[LOG_MAX_DATA_LEN];
945 strncpy(copy, message, LOG_MAX_DATA_LEN);
946 copy[LOG_MAX_DATA_LEN - 1] = '\0';
947
948 /* Remove colors */
949 log_remove_colors(copy);
950
951 /* Start Parsing */
952 int state = LOG_PARSER_INIT;
953 while (state < LOG_PARSER_DONE) {
954
955 switch (state) {
956 case LOG_PARSER_INIT: {
957 state = parse_datetime(copy, info);
958 break;
959 }
960 case LOG_PARSER_LEVEL: {
961 state = parse_level(copy, info);
962 break;
963 }
964 case LOG_PARSER_LOCATION: {
965 state = parse_location(copy, info);
966 break;
967 }
968 case LOG_PARSER_DATA: {
969 state = parse_data_string(message, size, copy);
970 break;
971 }
972 default: break;
973 }
974 }
975
976 /* Check parsing state */
977 return state == LOG_PARSER_DONE ? 0 : -1;
