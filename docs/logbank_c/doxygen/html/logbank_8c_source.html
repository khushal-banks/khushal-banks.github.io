197static int
198log_stamp_collector(char* stamp, const char* format, size_t size) {
199 if (stamp == NULL) {
200 /* stamp doesn't have a designated memory */
201 return -1;
202 }
203
204 /* Get raw-time */
205 time_t time_raw = time(NULL);
206
207 /* Receive time from assigned log_time_fn (this ensures correct time-zone) */
208 const struct tm* time_tm = log_time_fn != NULL ? log_time_fn(&time_raw)
209 : localtime(&time_raw);
210
211 /* Reformat time to requested format */
212 (void)strftime(stamp, size, format, (const struct tm*)time_tm);
213 stamp[size - 1] = '\0';
214 return 0;
215}

235static int
236update_header(char* header, size_t size, int level, log_locate_t locate) {
237 if (header == NULL) {
238 /* header doesn't have a dedicated memory */
239 return 0;
240 }
241
242 int len = 0;
243
244 /* dim colors of stamps */
245 len += LOG_ADD_COLOR(header + len, size - len, LOG_COLOR_GRAY);
246
247 /* log datestamps */
248 if (log_datestamp) {
249 char date_stamp[LOG_DATE_LEN];
250 log_stamp_collector(date_stamp, LOG_DATE_FORMAT, sizeof(date_stamp));
251
252 len += snprintf(header + len, size - len, "%s ", date_stamp);
253 }
254
255 /* log timestamps */
256 if (log_timestamp) {
257 char time_stamp[LOG_TIME_LEN];
258 log_stamp_collector(time_stamp, LOG_TIME_FORMAT, sizeof(time_stamp));
259
260 len += snprintf(header + len, size - len, "%s ", time_stamp);
261 }
262
263 /* use pre-defined color for level names */
264 len += LOG_ADD_COLOR(header + len, size - len, level);
265
266 /* Reading name of requested priority-level */
267 const char* name = log_get_levelname(level);
268
269 /* log priority-level name */
270 if (level == LOG_INFO || level == LOG_WARN) {
271 len += snprintf(header + len, size - len, "[%s] ", name);
272 } else {
273 len += snprintf(header + len, size - len, "[%s] ", name);
274 }
275
276 /* log location parameters */
277 if (locate.func != NULL && locate.file != NULL) {
278
279 /* dim color of location parameters and revert to color of log priority-level */
280 len += LOG_ADD_COLOR(header + len, size - len, LOG_COLOR_GRAY);
281 len += snprintf(header + len, size - len, "@ %s() \"%s +%d\" ", locate.func,
282 locate.file, locate.line);
283
284 len += LOG_ADD_COLOR(header + len, size - len, level);
285 }
286
287 /* log separator between header (message-prefix) and message */
288 len += snprintf(header + len, size - len, ": ");
289
290 /* terminate the use of colors */
291 len += LOG_ADD_COLOR(header + len, size - len, LOG_COLOR_END);
292
293 return len;
294}

314static void
315log_en_queue(const char* message) {
316 int read_index = log_queue.read_index;
317 int write_index = log_queue.write_index;
318 int queue_size = LOG_RETRIEVE_QUEUE_LEN + 1;
319
320 if ((write_index + 1) % queue_size == read_index) {
321 /* Queue Full */
322 log_queue.read_index = (read_index + 1) % queue_size;
323 }
324
325 char* rear = log_queue.data[write_index];
326 log_queue.write_index = (write_index + 1) % queue_size;
327
328 size_t size = LOG_MAX_DATA_LEN;
329 (void)snprintf(rear, size, "%s", message);
330 rear[size - 1] = '\0';
331}

342static int
343log_de_queue(char* message, size_t size) {
344 int read_index = log_queue.read_index;
345 int write_index = log_queue.write_index;
346 int queue_size = LOG_RETRIEVE_QUEUE_LEN + 1;
347
348 if (read_index == write_index) {
349 /* Queue Empty */
350 return 0;
351 }
352
353 const char* front = log_queue.data[read_index];
354 log_queue.read_index = (read_index + 1) % queue_size;
355
356 strncpy(message, front, size);
357 message[size - 1] = '\0';
358 return 1;
359}

459int
460log_this(int level, log_locate_t locate, const char* fmt, ...) {
461 int len = 0;
462
463 /* Filtering logic for lower priority log-requests and invalid priority requests */
464 if (level < log_level || level > LOG_FATAL) {
465 return len;
466 }
467
468 /* prepare log-message */
469 char message[LOG_MAX_DATA_LEN];
470 size_t size = sizeof(message);
471
472 /* Add header to log-message */
473 len += update_header(message, size - len, level, locate);
474
475 /* Add optional arguments to log-message */
476 va_list args;
477 va_start(args, fmt);
478 len += vsnprintf(message + len, size - len, fmt, args);
479 va_end(args);
480
481 /* Handle string termination properly */
482 len += snprintf(message + len, size - len, "\n");
483 message[size - 1] = '\0';
484
485 if (len > LOG_MAX_DATA_LEN) {
486 /* Data Overflow */
487 return (0 - len);
488 }
489
490 /* write operation */
491 log_stream_write(message);
492 log_en_queue(message);
493 return len;
494}
