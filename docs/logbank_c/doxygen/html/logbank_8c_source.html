266static int
267update_header(char* header, size_t size, int level, log_locate_t locate) {
268 if (header == NULL) {
269 /* header doesn't have a dedicated memory */
270 return 0;
271 }
272
273 int len = 0;
274
275 /* dim colors of stamps */
276 len += LOG_ADD_COLOR(header + len, size - len, LOG_COLOR_GRAY);
277
278 /* log datestamps */
279 if (log_datestamp) {
280 char date_stamp[LOG_DATE_LEN];
281 log_stamp_collector(date_stamp, LOG_DATE_FORMAT, sizeof(date_stamp));
282
283 len += snprintf(header + len, size - len, "%s ", date_stamp);
284 }
285
286 /* log timestamps */
287 if (log_timestamp) {
288 char time_stamp[LOG_TIME_LEN];
289 log_stamp_collector(time_stamp, LOG_TIME_FORMAT, sizeof(time_stamp));
290
291 len += snprintf(header + len, size - len, "%s ", time_stamp);
292 }
293
294 /* use pre-defined color for level names */
295 len += LOG_ADD_COLOR(header + len, size - len, level);
296
297 /* Reading name of requested priority-level */
298 const char* name = log_get_levelname(level);
299
300 /* log priority-level name */
301 if (level == LOG_INFO || level == LOG_WARN) {
302 len += snprintf(header + len, size - len, "[%s] ", name);
303 } else {
304 len += snprintf(header + len, size - len, "[%s] ", name);
305 }
306
307 /* log location parameters */
308 if (locate.func != NULL && locate.file != NULL) {
309
310 /* dim color of location parameters and revert to color of log priority-level */
311 len += LOG_ADD_COLOR(header + len, size - len, LOG_COLOR_GRAY);
312 len += snprintf(header + len, size - len, "@ %s() \"%s +%d\" ", locate.func,
313 locate.file, locate.line);
314
315 len += LOG_ADD_COLOR(header + len, size - len, level);
316 }
317
318 /* log separator between header (message-prefix) and message */
319 len += snprintf(header + len, size - len, ": ");
320
321 /* terminate the use of colors */
322 len += LOG_ADD_COLOR(header + len, size - len, LOG_COLOR_END);
323

346log_en_queue(const char* message) {
347 int read_index = log_queue.read_index;
348 int write_index = log_queue.write_index;
349 int queue_size = LOG_RETRIEVE_QUEUE_LEN + 1;
350
351 if ((write_index + 1) % queue_size == read_index) {
352 /* Queue Full */
353 log_queue.read_index = (read_index + 1) % queue_size;
354 }
355
356 char* rear = log_queue.data[write_index];
357 log_queue.write_index = (write_index + 1) % queue_size;
358
359 size_t size = LOG_MAX_DATA_LEN;
360 (void)snprintf(rear, size, "%s", message);

554parse_level(const char* copy, log_parser_info_t* info) {
555 int invalid = LOG_FATAL + 1;
556 info->level = invalid;
557
558 char name[] = "DEBUG";
559 size_t size = sizeof(name);
560
561 char* level_end = strchr(copy, ']');
562 if (level_end == NULL) {
563 return LOG_PARSER_FAILED;
564 }
565
566 level_end = level_end - 1;
567
568 const int max_name_len = 5;
569 char* level_start = level_end - max_name_len;
570
571 if (level_start[1] == '[') {
572 level_start = level_start + 1;

575 return LOG_PARSER_FAILED;

577
578 level_start = level_start + 1;
579 (void)snprintf(name, size, "%s", level_start);

731static int
732parse_data_string(char* message, size_t size, const char* copy) {
733 if (copy == NULL) {
734 /* No data available to parse */
735 return LOG_PARSER_FAILED;
736 }
737
738 /*
739 * Calculating Data Index
740 * HH:MM:SS [level] @ location : data
741 * 0 1 2
742 * */
743 int data_index = 2;
744
745 /* Separating header from copied data */
746 const char* dptr = strchr(copy, ':');
747 for (int count = 0; count < data_index && dptr != NULL; count++) {
748 dptr = strchr(dptr + 1, ':');
749 }
750
751 if (dptr == NULL) {
752 /* Time stamp is absent; Set data pointer to first ':' */
753 dptr = strchr(copy, ':');
754 }
755
756 strcpy(message, dptr + 2);
757 message[size - 1] = '\0';
758 return LOG_PARSER_DONE;
759}

837int
838log_this(int level, log_locate_t locate, const char* fmt, ...) {
839 int len = 0;
840
841 /* Filtering logic for lower priority log-requests and invalid priority requests */
842 if (level < log_level || level > LOG_FATAL) {
843 return len;
844 }
845
846 /* prepare log-message */
847 char message[LOG_MAX_DATA_LEN];
848 size_t size = sizeof(message);
849
850 /* Add header to log-message */
851 len += update_header(message, size - len, level, locate);
852
853 /* Add optional arguments to log-message */
854 va_list args;
855 va_start(args, fmt);
856 len += vsnprintf(message + len, size - len, fmt, args);
857 va_end(args);
858
859 /* Handle string termination properly */
860 len += snprintf(message + len, size - len, "\n");
861 message[size - 1] = '\0';
862
863 if (len > LOG_MAX_DATA_LEN) {
864 /* Data Overflow */
865 return (0 - len);
866 }
867
868 /* write operation */
869 log_stream_write(message);
870 log_en_queue(message);
871 return len;
872}

898int
899log_parser(char* message, size_t size, log_parser_info_t* info) {
900 if (message == NULL || size < LOG_MAX_DATA_LEN || info == NULL) {
901 /* Invalid parameters */
902 return -1;
903 }
904
905 char copy[LOG_MAX_DATA_LEN];
906 strcpy(copy, message);
907
908 log_remove_colors(copy);
909
910 int state = LOG_PARSER_INIT;
911 while (state < LOG_PARSER_DONE) {
912
913 switch (state) {
914 case LOG_PARSER_INIT: {
915 state = parse_datetime(copy, info);
916 break;
917 }
918 case LOG_PARSER_LEVEL: {
919 state = parse_level(copy, info);
920 break;
921 }
922 case LOG_PARSER_LOCATION: {
923 state = parse_location(copy, info);
924 break;
925 }
926 case LOG_PARSER_DATA: {
927 state = parse_data_string(message, size, copy);
928 break;
929 }
930 default: break;
931 }
932 }
933
934 return state == LOG_PARSER_DONE ? 0 : -1;
