359static int
360update_header(char* header, size_t size, int level, log_locate_t locate) {
361 if (header == NULL) {
362 /* header doesn't have a dedicated memory */
363 return 0;
364 }
365
366 int len = 0;
367
368 /* dim colors of stamps */
369 len += LOG_ADD_COLOR(header + len, size - len, LOG_COLOR_GRAY);
370
371 /* log datestamps */
372 if (log_datestamp) {
373 char date_stamp[LOG_DATE_LEN];
374 log_stamp_collector(date_stamp, LOG_DATE_FORMAT, sizeof(date_stamp));
375
376 len += snprintf(header + len, size - len, "%s ", date_stamp);
377 }
378
379 /* log timestamps */
380 if (log_timestamp) {
381 char time_stamp[LOG_TIME_LEN];
382 log_stamp_collector(time_stamp, LOG_TIME_FORMAT, sizeof(time_stamp));
383
384 len += snprintf(header + len, size - len, "%s ", time_stamp);
385 }
386
387 /* use pre-defined color for level names */
388 len += LOG_ADD_COLOR(header + len, size - len, level);
389
390 /* Reading name of requested priority-level */
391 const char* name = log_get_levelname(level);
392
393 /* log priority-level name */
394 if (level == LOG_INFO || level == LOG_WARN) {
395 len += snprintf(header + len, size - len, "[%s] ", name);
396 } else {
397 len += snprintf(header + len, size - len, "[%s] ", name);
398 }
399
400 /* log location parameters */
401 if (locate.func != NULL && locate.file != NULL) {
402
403 /* dim color of location parameters and revert to color of log priority-level */
404 len += LOG_ADD_COLOR(header + len, size - len, LOG_COLOR_GRAY);
405 len += snprintf(header + len, size - len, "@ %s() \"%s +%d\" ", locate.func,
406 locate.file, locate.line);
407
408 len += LOG_ADD_COLOR(header + len, size - len, level);
409 }
410
411 /* log separator between header (message-prefix) and message */
412 len += snprintf(header + len, size - len, ": ");
413
414 /* terminate the use of colors */
415 len += LOG_ADD_COLOR(header + len, size - len, LOG_COLOR_END);
416
417 return len;

1110int
1111log_this(int level, log_locate_t locate, const char* fmt, ...) {
1112 int len = 0;
1113
1114 /* Filtering logic for lower priority log-requests and invalid priority requests */
1115 if (level < log_level || level > LOG_FATAL) {
1116 return len;
1117 }
1118
1119 /* prepare log-message */
1120 char message[LOG_MAX_DATA_LEN];
1121 size_t size = sizeof(message);
1122
1123 /* Add header to log-message */
1124 len += update_header(message, size - len, level, locate);
1125
1126 /* Add optional arguments to log-message */
1127 va_list args;
1128 va_start(args, fmt);

1130 va_end(args);
1131

1143 log_en_queue(message);
