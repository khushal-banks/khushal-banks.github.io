130int
131snippet_parser(void) {
132 char message[LOG_MAX_DATA_LEN];
133 size_t size = sizeof(message);
134
135 /* Retrieve one log-message */
136 if (log_retrieve(message, size) == 0) {
137 return -1;
138 }
139
140 /* Parse retrieved log-message */
141 log_parser_info_t info;
142 int status = log_parser(message, size, &info);
143
144 /* print parsed info for proof of concept */
145 (void)fprintf(stderr, "date = %d/%d/%d\n", info.day, info.month, info.year);
146 (void)fprintf(stderr, "time = %d:%d:%d%s\n", info.hour, info.min, info.sec,
147 info.am_pm);
148 (void)fprintf(stderr, "level = %d) %s\n", info.level,
149 log_get_levelname(info.level));
150 (void)fprintf(stderr, "location = %s(): \"%s + %d\"\n", info.func, info.file,
151 info.line);
152 (void)fprintf(stderr, "data = %s\n", message);
153
154 return status;
155}
